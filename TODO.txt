  // All enemies get confused on hit.
  const confusionFrames = 5 * GAME_FRAMES_PER_SECOND;
  for (const npc of getNPCs()) {
    if (npc.IsVulnerableEnemy()) {
      npc.AddConfusion(EntityRef(player), confusionFrames, false);
